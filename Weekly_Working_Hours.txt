package weeklyWorkingHours;
import java.util.*;

public class WeeklyWorkingHours {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int X = sc.nextInt();
        int Y = sc.nextInt();
        int total = (4 * X) + Y;
        System.out.println(total);
        sc.close();
    }
}
